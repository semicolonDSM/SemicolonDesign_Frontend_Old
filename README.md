

# SemicolonDesign Frontend

## Install

```console
npm install --save @xquare/ui
yarn add @xquare/ui
```

## Usage

```jsx
import React, { Component } from 'react';

import { Button } from '@xquare/ui';

const Example: React.FC = () => {
    return(
        <Button>Hello, world!</Button>
    )
}
```
